// ---- 启动钩子 (Phase 6 T-6.2) ----
//
// RN/electron-shell 启动时调一次 initKbModule(), 幂等.
// 失败时 initState.error 暴露给 UI, UI 显示给用户 + 触发降级路径 (例如提示用户手动建目录).
//
// 设计要点:
//   - top-level 不 await (RN bundler 限制) → 必须显式调 initKbModule()
//   - 重复调用 idempotent (ensureKbDir 内部 mkdir recursive)
//   - 失败 throw → initState.error 兜底, UI 不白屏
//   - console.log KB_DIR_INFO 让 PM/verifier grep 验证 PRD 3.1 路径生效
//
// 业务代码:
//   import { initKbModule, getKbModuleInitState } from '...';
//   useEffect(() => { initKbModule().catch(() => { /* error already in state */ }); }, []);
//   const init = getKbModuleInitState();
//   if (init.error) return <ErrorScreen error={init.error} />;

import { ensureKbDir, kbDirInfo, getKbRoot } from './storage.ts';

export type KbModuleInitState = {
  /** initKbModule() 是否被调过至少一次 */
  initialized: boolean;
  /** 实际生效的 KB 根路径 (PRD 3.1 darwin 路径) */
  root: string | null;
  /** 最后一次 init 错误 (msg), null = 成功 */
  error: string | null;
  /** 最后一次 init 成功时间 (ISO) */
  ready_at: string | null;
};

const initState: KbModuleInitState = {
  initialized: false,
  root: null,
  error: null,
  ready_at: null,
};

/** 取启动 init 状态 (UI 渲染用) */
export function getKbModuleInitState(): Readonly<KbModuleInitState> {
  return initState;
}

/**
 * 启动钩子 — 幂等 ensure KB 路径存在 (PRD 3.1).
 *
 * 调用点:
 *   - Electron main process boot  (recommended: 在 app.whenReady 之后立刻调)
 *   - RN App.tsx root useEffect   (推荐: 第一次 render 后)
 *   - CLI 入口                     (cli/import-5-files-to-kb.ts)
 *
 * 失败时:
 *   - console.error 真实 err
 *   - initState.error 暴露
 *   - **不** 抛 — 业务调用方自己决定如何降级 (UI 显示 / CLI exit 1)
 *   - 但 ensureKbDir 内部会 throw, 这里用 try-catch 接住, 写 initState
 */
export async function initKbModule(opts: { rootOverride?: string } = {}): Promise<KbModuleInitState> {
  try {
    // 解决 root 早 expose (UI 在 await 完成前也能拿到目标路径)
    const expectedRoot = getKbRoot(opts.rootOverride);
    initState.root = expectedRoot;
    initState.initialized = true;
    initState.error = null;

    // 真正 ensure dir (递归建 kb/ + files/ + entries/)
    await ensureKbDir(opts.rootOverride);

    initState.ready_at = new Date().toISOString();
    initState.error = null;
    return initState;
  } catch (err) {
    const msg = `[initKbModule] KB init failed: ${(err as Error).message}`;
    console.error(msg);
    initState.error = msg;
    return initState;
  }
}

/** 给业务用的 debug helper — 直接 print kbDirInfo 对象 */
export function debugKbInfo(): ReturnType<typeof kbDirInfo> {
  return kbDirInfo();
}
