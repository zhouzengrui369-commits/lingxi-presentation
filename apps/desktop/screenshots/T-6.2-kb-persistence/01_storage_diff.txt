+/**
+ * ensureKbDir — startup 幂等确保 KB 根 + files/ + entries/ 3 个目录都存在。
+ * T-6.2 治本：app 第一次启动时调一次，后续 init() 也调 (idempotent)，失败 throw (不静默吞)。
+ *
+ * 用法:
+ *   import { ensureKbDir } from './storage';
+ *   await ensureKbDir();  // 不传 override → 走 PRD 3.1 路径
+ *
+ * 设计:
+ *   - mkdir recursive, 重复调用无副作用
+ *   - 失败 throw 真实 err, 业务层把它 wrap 成 UI init error 暴露给用户
+ *   - print KB_DIR_INFO 落 startup log (PM/verifier grep 用)
+ */
+export async function ensureKbDir(rootOverride?: string): Promise<string> {
+  const p = kbPaths(rootOverride);
+  try {
+    await fs.mkdir(p.root, { recursive: true });
+    await fs.mkdir(p.filesDir, { recursive: true });
+    await fs.mkdir(p.entriesDir, { recursive: true });
+  } catch (err) {
+    const msg = `[ensureKbDir] failed to create KB dir at ${p.root}: ${(err as Error).message}`;
+    console.error(msg);
+    throw new Error(msg);
+  }
+  // KB_DIR_INFO log (Phase 6 T-6.2 验收: grep 命中)
+  const info = kbDirInfo(rootOverride);
+  console.log(
+    `KB_DIR_INFO platform=${info.platform} root=${info.root} ` +
+      `files_dir=${info.files_dir} entries_dir=${info.entries_dir} ` +
+      `index_path=${info.index_path} manifest_path=${info.manifest_path}`,
+  );
+  return p.root;
+}

+export async function initKbModule(opts: { rootOverride?: string } = {}): Promise<KbModuleInitState> {
